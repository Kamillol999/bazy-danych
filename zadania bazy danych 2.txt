1.) SELECT employees.first_name, employees.last_name, SUM(salary_history.monthly_salary), AVG(salary_history.monthly_salary) FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id GROUP BY employees.employee_id;

2.)SELECT employees.first_name, employees.last_name, CONCAT(employees.city," ", employees.street_address,", ",employees.postal_code," ",employees.city," tel.",employees.phone_number) FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id GROUP BY employees.employee_id;

3.)SELECT employees.first_name, employees.last_name, avg(salary_history.monthly_salary), employees.hire_date FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id GROUP BY employees.employee_id ORDER BY employees.hire_date DESC;

4.)SELECT employees.first_name, employees.last_name, avg(salary_history.monthly_salary) as avg_salary FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id  GROUP BY employees.employee_id HAVING avg(salary_history.monthly_salary)>5500;

5.)SELECT employees.first_name, employees.last_name, YEAR(salary_history.salary_date), SUM(salary_history.monthly_salary) as avg_salary FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id  GROUP BY employees.employee_id, Year(salary_history.salary_date);

6.)SELECT employees.first_name, employees.last_name, YEAR(salary_history.salary_date), SUM(salary_history.monthly_salary) as avg_salary FROM employees JOIN salary_history ON employees.employee_id=salary_history.employee_id WHERE Year(salary_history.salary_date)=2020 GROUP BY employees.employee_id, Year(salary_history.salary_date) ORDER BY Sum(salary_history.monthly_salary) DESC Limit 1;
